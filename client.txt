import socket
TCP_IP = '127.0.0.1'
# hostname ipv6 address
# Each server listens for clients calling on one or more ports.
# A port may be a Fixnum port number, a string containing a port number, or the name of a service.
TCP_PORT = 5005
BUFFER_SIZE = 1024
MESSAGE = "Hello, World!"
# To create a socket, you must use the socket.socket() function available in socket module, which has the general syntax
# s = socket.socket (socket_family, socket_type, protocol=0)
s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
# in socket domain is socket.AF_INET
# The type of communications between the two endpoints, typically SOCK_STREAM for connection-oriented protocols
# the type of communication SOCK_DGRAM for connectionless protocols
s.connect((TCP_IP, TCP_PORT))
# client socket method
# s.connect() This method actively initiates TCP server connection.
byt = MESSAGE.encode('utf-8')
s.sendall(byt)
#s.sendto(byt, (TCP_IP, TCP_PORT))
while True:
    data = s.recv(BUFFER_SIZE)
    size = len(data)
    if size != 0:
        print(size)

# s.close() This method closes socket
s.close()
